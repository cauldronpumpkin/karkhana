### Clipboard fallback

Some browsers require HTTPS or user gesture for clipboard access. Add fallback:

```ts
async function copyText(text: string) {
  if (navigator.clipboard?.writeText) {
    return navigator.clipboard.writeText(text);
  }

  const textarea = document.createElement('textarea');
  textarea.value = text;
  textarea.style.position = 'fixed';
  textarea.style.opacity = '0';
  document.body.appendChild(textarea);
  textarea.select();
  document.execCommand('copy');
  document.body.removeChild(textarea);
}
```

### Rendering helpers

```python
def _format_context_files(files: list[ContextFileRef]) -> str:
    if not files:
        return "- No specific context files were provided. Inspect the repository structure first."
    return "\n".join(
        f"- `{f.path}`" + (f" — {f.reason}" if f.reason else "")
        for f in files
    )


def _format_list(items: list[str], fallback: str) -> str:
    if not items:
        return f"- {fallback}"
    return "\n".join(f"- {item}" for item in items)


def _format_commands(commands: list[str]) -> str:
    if not commands:
        return "- No verification commands were provided. Identify and run the relevant test/check commands."
    return "\n".join(f"- `{cmd}`" for cmd in commands)
```

### Step 1: Code terminology migration

### Step 2: API compatibility period

Keep old endpoints temporarily.

In `backend/app/routers/build.py`:

```python
@router.post("/projects/{project_id}/prometheus", deprecated=True)
async def generate_prometheus_alias(
    project_id: str,
    body: GenerateCodexTaskRequest,
    response: Response,
):
    response.headers["Deprecation"] = "true"
    response.headers["Link"] = f"</build/projects/{project_id}/codex-task>; rel=\"successor-version\""
    return await generate_codex_task(project_id, body)
```

This avoids breaking existing frontend or worker bridge clients immediately.

---

### Step 3: Stored memory event migration

If existing memory events use:

```python
event_type = "prometheus_prompt_generated"
```

migrate to:

```python
event_type = "codex_task_generated"
```

Recommended database migration script:

```python
UPDATE memory_events
SET event_type = 'codex_task_generated'
WHERE event_type = 'prometheus_prompt_generated';
```

If payload contains:

```json
{
  "prometheus_prompt": "..."
}
```

transform to:

```json
{
  "task": {
    "prompt": "...",
    "status": "generated"
  },
  "legacy": {
    "source_event_type": "prometheus_prompt_generated"
  }
}
```

If the storage is document-based, migration pseudocode:

```python
async def migrate_prometheus_events(memory_store):
    events = await memory_store.find_events(event_type="prometheus_prompt_generated")

    for event in events:
        payload = event.payload or {}

        prompt = (
            payload.get("prometheus_prompt")
            or payload.get("prompt")
            or payload.get("handoff")
            or ""
        )

        task = {
            "id": payload.get("id") or f"legacy_codex_task_{event.id}",
            "project_id": event.project_id,
            "work_item_id": event.work_item_id,
            "title": payload.get("title") or "Legacy Codex task",
            "goal": payload.get("goal") or "Legacy generated task",
            "repo_path": payload.get("repo_path") or "",
            "context_files": payload.get("context_files") or [],
            "constraints": payload.get("constraints") or [],
            "deliverables": payload.get("deliverables") or [],
            "verification_commands": payload.get("verification_commands") or [],
            "prompt": prompt,
            "status": "generated",
            "created_at": event.created_at,
            "updated_at": None,
        }

        await memory_store.update_event(
            event.id,
            event_type="codex_task_generated",
            payload={
                "task": task,
                "legacy": {
                    "source_event_type": "prometheus_prompt_generated",
                    "original_payload": payload,
                },
            },
        )
```
