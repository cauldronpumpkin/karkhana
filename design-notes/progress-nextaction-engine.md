# Implementation Plan: Progress Dashboard + Next-Action Engine for Karkhana

## Goal

Build a crisp “what should I work on next?” experience for Karkhana.

The system should generate a per-idea/project progress summary by detecting blockers from:

- Incomplete phases
- Missing scores
- Pending research tasks
- Stale Project Twin indices
- Failed work items
- No worker connected
- Incomplete build handoff

Then it should recommend ranked next actions with:

- Priority
- Reason
- Suggested owner
- Optional Codex-ready prompt
- Related entity references
- Completion status

---

async def compute_progress_summary(
    idea_id: str,
    persist: bool = True,
) -> ProgressSummary:
    source = await load_progress_source_data(idea_id)

    phase_progress = compute_phase_progress(source.phases)
    score_progress = compute_score_progress(source.scores)
    research_progress = compute_research_progress(source.research_tasks)
    project_twin_progress = compute_project_twin_progress(source.project_twin)
    worker_progress = compute_worker_progress(source.workers)
    build_handoff_progress = compute_build_handoff_progress(source.build_handoff)

    blockers = detect_blockers(
        source=source,
        phase_progress=phase_progress,
        score_progress=score_progress,
        research_progress=research_progress,
        project_twin_progress=project_twin_progress,
        worker_progress=worker_progress,
        build_handoff_progress=build_handoff_progress,
    )

    actions = generate_next_actions(
        source=source,
        blockers=blockers,
        phase_progress=phase_progress,
        score_progress=score_progress,
        research_progress=research_progress,
        project_twin_progress=project_twin_progress,
        worker_progress=worker_progress,
        build_handoff_progress=build_handoff_progress,
    )

    actions = apply_persisted_action_overrides(
        computed_actions=actions,
        persisted_actions=source.persisted_actions,
    )

    actions = rank_next_actions(actions, blockers, source)

    progress_percent = compute_progress_percent(
        phase_progress=phase_progress,
        score_progress=score_progress,
        research_progress=research_progress,
        build_handoff_progress=build_handoff_progress,
    )

    overall_status = compute_overall_status(
        blockers=blockers,
        progress_percent=progress_percent,
        phase_progress=phase_progress,
        build_handoff_progress=build_handoff_progress,
        source=source,
    )

    summary = ProgressSummary(
        idea_id=idea_id,
        project_id=source.idea.get("project_id"),
        overall_status=overall_status,
        progress_percent=progress_percent,
        phase_progress=phase_progress,
        score_progress=score_progress,
        research_progress=research_progress,
        project_twin_progress=project_twin_progress,
        worker_progress=worker_progress,
        build_handoff_progress=build_handoff_progress,
        blockers=blockers,
        next_actions=actions,
        computed_at=datetime.utcnow(),
    )

    if persist:
        await save_progress_summary(summary)

    return summary
```

def detect_blockers(
    source: ProgressSourceData,
    phase_progress: dict[str, PhaseProgressItem],
    score_progress: ScoreProgress,
    research_progress: ResearchProgress,
    project_twin_progress: ProjectTwinProgress,
    worker_progress: WorkerProgress,
    build_handoff_progress: BuildHandoffProgress,
) -> list[ProgressBlocker]:
    blockers = []

    blockers.extend(detect_incomplete_phase_blockers(phase_progress))
    blockers.extend(detect_missing_score_blockers(score_progress))
    blockers.extend(detect_pending_research_blockers(source.research_tasks))
    blockers.extend(detect_stale_project_index_blockers(project_twin_progress))
    blockers.extend(detect_failed_work_item_blockers(source.work_items))
    blockers.extend(detect_no_worker_connected_blockers(source.work_items, worker_progress))
    blockers.extend(detect_incomplete_build_handoff_blockers(build_handoff_progress))

    return dedupe_blockers(blockers)
```

### 2.5.1 Incomplete Phase Detection

```python
def detect_incomplete_phase_blockers(
    phase_progress: dict[str, PhaseProgressItem],
) -> list[ProgressBlocker]:
```

Rules:

- If the first incomplete phase appears before `build`, create blocker.
- Severity:
  - `high` for incomplete `capture`, `clarify`, `plan`, `validate`
  - `medium` for incomplete `research`, `score`, `shape`
  - `critical` for incomplete `build` if all prior phases are complete

Use deterministic blocker IDs:

```python
blocker_id = f"incomplete_phase:{phase}"
```

Example blocker:

```python
ProgressBlocker(
    blocker_id="incomplete_phase:plan",
    blocker_type=BlockerType.INCOMPLETE_PHASE,
    severity=BlockerSeverity.HIGH,
    title="Plan phase is incomplete",
    description="The idea is missing an implementation plan before it can move confidently into build.",
    phase="plan",
    related_entity_type="phase",
    related_entity_id="plan",
    detected_at=now,
)
```

Important: Do not generate eight phase blockers. Generate at most the next incomplete phase blocker plus explicit blockers for phases marked `blocked`.

## 3.3 Router Implementation Skeleton

```python
from fastapi import APIRouter, Query
from app.services.progress.engine import compute_progress_summary
from app.services.progress.repository import update_action_status

router = APIRouter(tags=["progress"])


@router.get("/ideas/{idea_id}/progress")
async def get_idea_progress(
    idea_id: str,
    refresh: bool = Query(False),
):
    return await compute_progress_summary(idea_id=idea_id, persist=True)


@router.post("/ideas/{idea_id}/progress/refresh")
async def refresh_idea_progress(idea_id: str):
    summary = await compute_progress_summary(idea_id=idea_id, persist=True)
    return {"summary": summary}


@router.get("/ideas/{idea_id}/next-actions")
async def get_idea_next_actions(
    idea_id: str,
    refresh: bool = Query(False),
    limit: int = Query(5, ge=1, le=25),
    include_dismissed: bool = Query(False),
):
    summary = await compute_progress_summary(idea_id=idea_id, persist=True)
    actions = summary.next_actions

    if not include_dismissed:
        actions = [a for a in actions if a.status != "dismissed"]

    return {
        "idea_id": idea_id,
        "actions": actions[:limit],
    }


@router.patch("/ideas/{idea_id}/next-actions/{action_id}")
async def patch_next_action(
    idea_id: str,
    action_id: str,
    payload: UpdateNextActionRequest,
):
    action = await update_action_status(
        idea_id=idea_id,
        action_id=action_id,
        status=payload.status,
        metadata=payload.metadata,
    )
    return {"action": action}
```

Note: The first endpoint should respect cache in final implementation. The skeleton recomputes every time for simplicity.

  ideaId: string,

): Promise<NextAction[]> {
  const res = await fetch(
    `/api/ideas/${ideaId}/next-actions?limit=${limit}&refresh=${refresh}`
  );
  if (!res.ok) throw new Error('Failed to load next actions');
  const data = await res.json();

export async function updateNextActionStatus(
  ideaId: string,

# 6. Priority Ranking Algorithm: Heuristic for Ranking Actions

File:

```text
backend/app/services/progress/ranking.py
```

## 6.1 Ranking Goals

The top action should usually be:

1. The earliest unresolved critical blocker.
2. Something the user can actually do now.
3. Something that unlocks many downstream actions.
4. Something aligned with the idea’s current phase.
5. A build-critical operational fix if build is ready but blocked.

---

## 6.2 Priority Score Formula

Each action receives:

```python
priority_score =
    severity_score
    + phase_score
    + unblock_score
    + readiness_score
    + owner_score
    + staleness_score
    - dependency_penalty
    - dismissed_penalty
```

Clamp to:

```python
0 <= priority_score <= 100
```

---

## 6.3 Component Scores

### Severity Score

Based on related blockers:

```python
SEVERITY_POINTS = {
    "critical": 45,
    "high": 35,
    "medium": 22,
    "low": 10,
}
```

Use maximum related blocker severity.

If no blocker but proactive action:

```python
severity_score = 8
```

---

### Phase Score

Earlier incomplete phases get higher priority.

```python
PHASE_POINTS = {
    "capture": 28,
    "clarify": 26,
    "research": 22,
    "score": 22,
    "shape": 20,
    "plan": 24,
    "validate": 24,
    "build": 26,
}
```

Reasoning:

- Capture/clarify unblock everything.
- Plan/validate/build are highly execution-relevant.
- Research/score/shape are important but sometimes parallelizable.

---

### Unblock Score

Based on number of downstream phases or systems unlocked.

```python
unblock_score = min(20, 3 * number_of_downstream_dependencies)
```

Examples:

- Complete Plan may unlock Validate + Build: `6`
- Connect Worker may unlock multiple queued work items: up to `20`
- Reindex Project Twin unlocks safe coding context: `12`

---

### Readiness Score

Add if action is executable immediately.

```python
if action.suggested_owner == "human":
    readiness_score = 8
elif action.suggested_owner == "system" and required_api_exists:
    readiness_score = 10
elif action.suggested_owner == "local_worker" and worker_available:
    readiness_score = 12
elif action.suggested_owner == "local_worker" and not worker_available:
    readiness_score = -10
else:
    readiness_score = 5
```

---

### Owner Score

Prefer human-actionable next steps slightly in dashboard.

```python
OWNER_POINTS = {
    "human": 7,
    "system": 6,
    "ai_agent": 5,
    "local_worker": 4,
}
```

Exception:

- If idea is ready for build and worker available, local worker actions get `+10`.

---

### Staleness Score

Add urgency for old blockers.

```python
age_hours = hours_since_oldest_related_blocker_detected
staleness_score = min(10, age_hours / 24)
```

---

### Dependency Penalty

Penalize actions blocked by prerequisites.

Examples:

- `START_BUILD` while handoff incomplete: `-40`
- `RETRY_WORK_ITEM` with no worker connected: `-25`
- `REINDEX_PROJECT` while no Project Twin exists: `-10`
- `RUN_RESEARCH` while capture incomplete: `-15`

---

### Dismissed Penalty

If action status is dismissed:

```python
dismissed_penalty = 100
```

Usually dismissed actions are filtered out, but if included, they should sink.

---

## 6.4 Priority Labels

```python
def priority_label(score: float) -> str:
    if score >= 80:
        return "urgent"
    if score >= 60:
        return "high"
    if score >= 35:
        return "medium"
    return "low"
```

---

## 6.5 Deterministic Tie-Breaking

Sort by:

```python
(
    -priority_score,
    phase_order_index,
    action_type_order_index,
    action_id,
)
```

Action type order:

```python
ACTION_TYPE_ORDER = [
    "connect_worker",
    "retry_work_item",
    "complete_phase",
    "complete_build_handoff",
    "reindex_project",
    "run_research",
    "review_research",
    "score_idea",
    "start_build",
    "review_build_output",
]
```

But apply context override:

- If not in build phase, `complete_phase` should beat `connect_worker`.
- If build work queued, `connect_worker` should beat earlier non-critical recommendations.

---

## 6.6 Function Skeleton

```python
def rank_next_actions(
    actions: list[NextAction],
    blockers: list[ProgressBlocker],
    source: ProgressSourceData,
) -> list[NextAction]:
    blocker_by_id = {b.blocker_id: b for b in blockers}

    ranked = []
    for action in actions:
        score = compute_action_priority_score(
            action=action,
            blocker_by_id=blocker_by_id,
            source=source,
        )
        action.priority_score = score
        action.priority_label = priority_label(score)
        ranked.append(action)

    return sorted(
        ranked,
        key=lambda a: (
            -a.priority_score,
            phase_sort_index(a.phase),
            action_type_sort_index(a.action_type),
            a.action_id,
        ),
    )
```
